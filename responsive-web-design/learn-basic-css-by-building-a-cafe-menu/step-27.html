Чтобы применить стиль класса к div элементу, удалите id атрибут и добавьте атрибут class к открывающему тегу элемента div. Убедитесь, что для значения класса установлено значение menu.

Файл index.html 
      
        <!DOCTYPE html>
        <html lang="ru">
          <head>
            <title>Меню кафе</title>
            <meta charset="utf-8">
            <meta name="viewport" content="width=device-width, initial-scale=1.0" />
            <link rel="stylesheet" href="styles.css">
          </head>
          <body>
            <div id="menu">
              <main>
              <h1>КЕМПЕР КАФЕ</h1>
              <p>Основано в 2020</p>
              <section>
                <h2>Кофе</h2>
              </section>
              </main>
            </div>
          </body>
        </html>
Файл styles.css 
      
        h1, h2, p {
          text-align: center;
        }

        body {
          /* background-color: burlywood; */
        }

        .menu {
          width: 80%;
          background-color: burlywood;
          margin-left: auto;
          margin-right: auto;
        }
