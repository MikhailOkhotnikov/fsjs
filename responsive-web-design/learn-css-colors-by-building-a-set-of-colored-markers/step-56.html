        <html lang="ru">
          <head>
            <meta charset="utf-8">

            <link rel="stylesheet" href="styles.css">
            <title>Цветные маркеры</title>      
          </head>
          <body>
            <h1>Цветные маркеры CSS</h1>
            <div class="container">
              <div class="marker red"></div>
              <div class="marker green"></div>
              <div class="marker blue"></div>
            </div>
          </body>
        </html>
