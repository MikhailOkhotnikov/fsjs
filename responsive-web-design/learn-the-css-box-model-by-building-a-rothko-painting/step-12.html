Файл index.html

        <html lang="ru">
          <head>
            <meta charset="UTF-8">
            <link rel="stylesheet" href="styles.css">
            <title>Картина Ротко</title>
          </head>
          <body>
            <div class="frame">
              <div class="canvas"></div>
            </div>
          </body>
        </html>

        .canvas {
          width: 500px;
          height: 600px;
          background-color: #4d0f00;
        }
